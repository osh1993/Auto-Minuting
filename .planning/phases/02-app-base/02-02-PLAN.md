1. data/local/entity/MeetingEntity.kt:
   - @Entity(tableName = "meetings")
   - @PrimaryKey(autoGenerate = true) val id: Long = 0
   - val title: String, val recordedAt: Long (epoch millis), val audioFilePath: String
   - val transcriptPath: String? = null, val minutesPath: String? = null
   - val pipelineStatus: String (enum name으로 저장)
   - val errorMessage: String? = null
   - val createdAt: Long, val updatedAt: Long
   - fun toDomain(): Meeting 매핑 함수
   - companion object { fun fromDomain(meeting: Meeting): MeetingEntity } 매핑 함수

2. data/local/converter/Converters.kt:
   - @TypeConverter fun fromPipelineStatus(status: PipelineStatus): String = status.name
   - @TypeConverter fun toPipelineStatus(value: String): PipelineStatus = PipelineStatus.valueOf(value)
   - @TypeConverter fun fromInstant(instant: Instant): Long = instant.toEpochMilli()
   - @TypeConverter fun toInstant(value: Long): Instant = Instant.ofEpochMilli(value)

3. data/local/dao/MeetingDao.kt:
   - @Dao interface MeetingDao
   - @Query("SELECT * FROM meetings ORDER BY recordedAt DESC") fun getAllMeetings(): Flow<List<MeetingEntity>>
   - @Query("SELECT * FROM meetings WHERE id = :id") fun getMeetingById(id: Long): Flow<MeetingEntity?>
   - @Insert(onConflict = OnConflictStrategy.REPLACE) suspend fun insert(entity: MeetingEntity): Long
   - @Update suspend fun update(entity: MeetingEntity)
   - @Query("UPDATE meetings SET pipelineStatus = :status, errorMessage = :errorMessage, updatedAt = :updatedAt WHERE id = :id") suspend fun updatePipelineStatus(id: Long, status: String, errorMessage: String?, updatedAt: Long)
   - @Query("DELETE FROM meetings WHERE id = :id") suspend fun delete(id: Long)

4. data/local/AppDatabase.kt:
   - @Database(entities = [MeetingEntity::class], version = 1, exportSchema = false)
   - @TypeConverters(Converters::class)
   - abstract class AppDatabase : RoomDatabase()
   - abstract fun meetingDao(): MeetingDao
